   def check (objA,objB):

       tmp := @(objA) - @(objB).

       distribute     (tmp)
       sort_product   (tmp)
       rename_dummies (tmp)
       canonicalise   (tmp)

       return tmp

   import cdblib

   json000A = cdblib.get ('json000','../expected/example-04.json')

   # ---------------------------------------------
   diff000 = check (json000A,json000B)   # cdb (ex-04.diff000,diff000)
